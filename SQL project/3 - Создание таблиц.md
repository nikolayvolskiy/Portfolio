```sql
-- 1. КАТЕГОРИИ ТОВАРОВ
CREATE TABLE categories (
    category_id TEXT PRIMARY KEY,
    category_name TEXT
);

-- 2. ТОВАРЫ
CREATE TABLE items (
    item_id TEXT PRIMARY KEY,
    item_code TEXT,
    item_name TEXT,
    category_id TEXT,
    color_name TEXT,
    size_name TEXT,
    season_name TEXT,
    FOREIGN KEY (category_id) REFERENCES categories(category_id)
);

-- 3. МАГАЗИНЫ
CREATE TABLE stores (
    store_id TEXT PRIMARY KEY,
    store_name TEXT,
    store_category TEXT,
    store_direction TEXT
);

-- 4. СОТРУДНИКИ
CREATE TABLE employees (
    employee_id TEXT PRIMARY KEY,
    employee_name TEXT
);

-- 5. ТРАФИК МАГАЗИНА
CREATE TABLE traffic (
    date TEXT,
    traffik_entrance INTEGER,
    traffic_exit INTEGER,
    traffic_average REAL
);

-- 6. ОСНОВНЫЕ ПРОДАЖИ (оффлайн)
-- sales_transaction_id НЕ уникальный - в одном чеке может быть несколько товаров
CREATE TABLE main_sales (
    sales_transaction_id TEXT,
    sales_time TEXT,
    store_id TEXT,
    customer_phone TEXT,
    order_number_store TEXT,
    order_number_site TEXT,
    employee_id TEXT,
    item_id TEXT,
    qty INTEGER,
    total_value REAL,
    FOREIGN KEY (store_id) REFERENCES stores(store_id),
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
    FOREIGN KEY (item_id) REFERENCES items(item_id)
);

-- 7. ПРОДАЖИ ПАРТНЕРОВ (комиссионные)
CREATE TABLE com_sales (
    sales_transaction_id TEXT,
    sales_time TEXT,
    store_id TEXT,
    customer_phone TEXT,
    order_number_store TEXT,
    order_number_site TEXT,
    employee_id TEXT,
    item_id TEXT,
    qty INTEGER,
    total_value REAL,
    FOREIGN KEY (store_id) REFERENCES stores(store_id),
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
    FOREIGN KEY (item_id) REFERENCES items(item_id)
);

-- 8. ОПТОВЫЕ ПРОДАЖИ (B2B)
CREATE TABLE b2b_sales (
    sales_transaction_id TEXT,
    sales_time TEXT,
    store_id TEXT,
    customer_phone TEXT,
    order_number_store TEXT,
    order_number_site TEXT,
    employee_id TEXT,
    item_id TEXT,
    qty INTEGER,
    total_value REAL,
    FOREIGN KEY (store_id) REFERENCES stores(store_id),
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
    FOREIGN KEY (item_id) REFERENCES items(item_id)
);

-- 9. ВОЗВРАТЫ ОФФЛАЙН
CREATE TABLE main_returns (
    return_transaction_id TEXT,
    return_time TEXT,
    store_id TEXT,
    customer_phone TEXT,
    employee_id TEXT,
    item_id TEXT,
    qty INTEGER,
    total_value REAL,
    sales_transaction_id TEXT,
    sales_time TEXT,
    FOREIGN KEY (store_id) REFERENCES stores(store_id),
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
    FOREIGN KEY (item_id) REFERENCES items(item_id)
);

-- 10. ВОЗВРАТЫ ПАРТНЕРОВ
CREATE TABLE com_returns (
    return_transaction_id TEXT,
    return_time TEXT,
    store_id TEXT,
    customer_phone TEXT,
    employee_id TEXT,
    item_id TEXT,
    qty INTEGER,
    total_value REAL,
    sales_transaction_id TEXT,
    sales_time TEXT,
    FOREIGN KEY (store_id) REFERENCES stores(store_id),
    FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
    FOREIGN KEY (item_id) REFERENCES items(item_id)
);
```
